314  template<typename Target>
315  inline void parseProperties(eaglexml::xml_node<>* firstProp,
316  DataMap<CachedString, CachedString>* tagMap,
317  std::vector<NodeId>* nodeRefIds,
318  DataMap<NodeId, CachedString>* nodeRoleMap,
319  std::vector<WayId>* wayRefIds,
320  DataMap<WayId, CachedString>* wayRoleMap)
321  {
322  BOOST_STATIC_ASSERT((boost::is_same<Target, Node>::value || boost::is_same<Target, Way>::value || boost::is_same<Target, Relation>::value));
323  assert(tagMap);
324 
325  // Go through all properties and parse them
326  eaglexml::xml_node<>* prop = firstProp;
327  for(;prop; prop = prop->next_sibling())
328  {
329  const char* propName = firstProp->name();
330 
331  if(std::strcmp(propName, "tag") == 0)
332  {
333  // Parse tag
334  CachedString key;
335  CachedString value;
336  extractAttributeFromNode("k", prop, &key);
337  extractAttributeFromNode("v", prop, &value);
338 
339  (*tagMap)[key] = value;
340 
341  }else if(boost::is_same<Target, Way>::value && std::strcmp(propName, "nd") == 0)
342  {
343  assert(nodeRefIds);
344 
345  // Parse a node reference
346  OsmIdType osmId;
347  extractAttributeFromNode("ref", prop, &osmId);
348  nodeRefIds->push_back(resolveOsmId(osmId, nodeIdMapping));
349  }else if(boost::is_same<Target, Relation>::value && std::strcmp(propName, "member") == 0)
350  {
351  // Parse a relation member
352  assert(nodeRefIds);
353  assert(nodeRoleMap);
354  assert(wayRefIds);
355  assert(wayRoleMap);
356 
357 
358  eaglexml::xml_attribute<>* attr = prop->first_attribute("type");
359 
360  if(!attr || attr->value_size() == 0)
361  BOOST_THROW_EXCEPTION(excp::InputFormatException() << excp::InfoXmlEntityName("relation::member::type"));
362 
364  OsmIdType osmRefId;
365  extractAttributeFromNode("ref", prop, &osmRefId);
366 
367  // Evaluate type attribute
368  const char* type = attr->value();
369  if(std::strcmp(type, "node") == 0)
370  {
371  NodeId nodeId = resolveOsmId(osmRefId, nodeIdMapping);
372  nodeRefIds->push_back(nodeId);
373  extractAttributeFromNode("role", prop, &((*nodeRoleMap)[nodeId]));
374  }else if(std::strcmp(type, "way") == 0)
375  {
376  WayId wayId = resolveOsmId(osmRefId, wayIdMapping);
377  wayRefIds->push_back(wayId);
378  extractAttributeFromNode("role", prop, &((*wayRoleMap)[wayId]));
379  }else if(std::strcmp(type, "relation") == 0) {
380 
381  if(!outputIgnoreRelation)
382  {
383  LOG_SEV(importer_log, warning) << "This software does not support relation member in relations!";
384  LOG_SEV(importer_log, warning) << "Reference is ignored!";
385  outputIgnoreRelation = true;
386  }
387  continue;
388  }else{
389 
390  BOOST_THROW_EXCEPTION(excp::InputFormatException() << excp::InfoXmlEntityName("relation::member::type::value"));
391  }
392 
393  }
394  }
